<accordion
	timing="200"
	handle="myaccordion"
	close-others="false"
	toggle-icon="chevron">
	<toggle ng-repeat="c in content" content="c.value"></toggle>
</accordion>

<button ng-click="myaccordion.expand()">expand all</button>
<button ng-click="myaccordion.collapse()">collapse all</button>
<button ng-click="myaccordion.expand(1)">expand index 1</button>
<button ng-click="myaccordion.collapse(1)">collapse index 1</button>
<button ng-click="myaccordion.disable()">disable all</button>
<button ng-click="myaccordion.enable()">enable all</button>
<button ng-click="myaccordion.disable(1)">disable 1</button>	
<button ng-click="myaccordion.enable(1)">enable 1</button>
